1. 객체를 사용해 도형 그리기

도형을 객체로 만들고, 캔버스에 그리는 방식


class Rectangle {
  constructor(x, y, width, height, color) {
    this.x = x;
    this.y = y;
    this.width = width;
    this.height = height;
    this.color = color;
  }

  draw(ctx) {
    ctx.fillStyle = this.color;
    ctx.fillRect(this.x, this.y, this.width, this.height);
  }
}

const canvas = document.getElementById('myCanvas');
const ctx = canvas.getContext('2d');

const rect = new Rectangle(20, 20, 100, 50, 'blue');
rect.draw(ctx);

2. 애니메이션 효과 추가하기

requestAnimationFrame을 이용해 반복 그리기


let x = 0;
function animate() {
  ctx.clearRect(0, 0, canvas.width, canvas.height);

  rect.x = x;
  rect.draw(ctx);

  x += 2;
  if (x > canvas.width) x = -rect.width;

  requestAnimationFrame(animate);
}

animate();
